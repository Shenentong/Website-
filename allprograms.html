import java.awt.*;
import javax.swing.*;

public class ChineseFlag extends Canvas
{
   
    // Method to draw the five-pointed stars
    private void drawStar(Graphics g, int centerX, int centerY, int size)
    {
       g.setColor(Color.YELLOW);
       int[] xPoints = new int[10];
       int[] yPoints = new int[10];
      
       // Coordinates of the five-pointed star
       double angle = Math.PI / 5;
       for (int i = 0; i < 5; i++)
       {
          // Outer vertices
          xPoints[i * 2] = (int) (centerX + size * Math.cos(i * 2 * angle));
          yPoints[i * 2] = (int) (centerY - size * Math.sin(i * 2 * angle));
          // Inner vertices
          xPoints[i * 2 + 1] = (int) (centerX + size / 2.5 * Math.cos(i * 2 * angle + angle));
          yPoints[i * 2 + 1] = (int) (centerY - size / 2.5 * Math.sin(i * 2 * angle + angle));
       }
      
       // Draw the star
       g.fillPolygon(xPoints, yPoints, 10);
    }
   
    @Override
    public void paint(Graphics g)
    {
       // Set the background color of the flag
       g.setColor(Color.RED);
       g.fillRect(0, 0, 800, 480); // The flag has a width of 800px and height of 480px
      
       // Draw the large star (first star)
       drawStar(g, 100, 80, 50); // Coordinates and size
      
       // Draw the four smaller stars
       drawStar(g, 180, 40, 20); // Top-right star
       drawStar(g, 210, 125, 20); // Bottom-right star
       drawStar(g, 230, 80, 20); // Right-center star
       drawStar(g, 160, 160, 20); // Bottom-right star
      
       // Optionally add additional shapes or annotations if necessary
      
    }
   
    public static void main(String[] args)
    {
       // Create the JFrame to display the flag
       JFrame frame = new JFrame( "Chinese National Flag" );
       Canvas canvas = new ChineseFlag();
      
       // Set up the JFrame settings
       frame.setSize(800, 480); // Set the size of the flag
       frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
       frame.add(canvas); // Add the canvas to the frame
       frame.setVisible(true); // Make the window visible
      
    }
}

import java.util.Scanner;

public class NumberSumScanner
{
    public static void main(String[] args)
    {
       Scanner scanner = new Scanner(System.in);
       double sum = 0;
       boolean hasNonNumber = false;
      
       System.out.println( "Please enter numbers separated by whitespace (press Enter when done):" );
      
       while (scanner.hasNext())
       {
          if (scanner.hasNextDouble())
          {
             double num = scanner.nextDouble();
             sum += num;
          }
          else
          {
             // If it's not a number, mark that non-number input has been encountered
             hasNonNumber = true;
             scanner.next(); // Consume the non-number token to move to the next input
            
          }
       }
      
       if (hasNonNumber)
       {
          System.out.println( "NaN" );
       }
       else
       {
          System.out.println( "The sum of the entered numbers is: " + sum);
       }
      
       scanner.close();
    }
}

import java.io.File;
import java.util.Scanner;
import java.io.FileNotFoundException;

public class DataAnalysisProgram
{
    public static void main(String[] args)
    {
       StringBuilder storestring = new StringBuilder();
       try
       {
          // 修改文件路径为指定的绝对路径
          Scanner s = new Scanner(new File( "/Users/shenentong/Desktop/data.txt" ));
          int n = 0;
          double idsum = 0, Gradesum = 0, Scoresum = 0, Absencesum = 0, Timesum = 0, Startsum = 0, Subsum = 0, Quizsum = 0, Ordersum = 0;
          double idmean = 0, Grademean = 0, Scoremean = 0, Absencemean = 0, Timemean = 0, Startmean = 0, Submean = 0, Quizmean = 0, Ordermean = 0;
          double iddev = 0, Gradedev = 0, Scoredev = 0, Absencedev = 0, Timedev = 0, Startdev = 0, Subdev = 0, Quizdev = 0, Orderdev = 0;
          String id, Grade, Score, Absence, Time, Start, Sub, Quiz, Order;
          s.useDelimiter( "\t|\n" );
         
          while (s.hasNext())
          {
             n++;
             id = s.next().trim();
             Grade = s.next().trim();
             Score = s.next().trim();
             Absence = s.next().trim();
             Time = s.next().trim();
             Start = s.next().trim();
             Sub = s.next().trim();
             Quiz = s.next().trim();
             s.useDelimiter( "\n" );
             Order = s.next().trim();
             s.useDelimiter( "\t" );
            
             storestring.append(id).append( "\t" );
             storestring.append(Grade).append( "\t" );
             storestring.append(Score).append( "\t" );
             storestring.append(Absence).append( "\t" );
             storestring.append(Time).append( "\t" );
             storestring.append(Start).append( "\t" );
             storestring.append(Sub).append( "\t" );
             storestring.append(Quiz).append( "\t" );
             storestring.append(Order).append( "\t" );
            
             if (n == 1)
             {
                continue;
             }
            
             idsum += Double.parseDouble(id);
             double GradeNum = 0;
             if ( "A" .equals(Grade))
             {
                GradeNum = 4;
             }
             else if ( "B" .equals(Grade))
             {
                GradeNum = 3;
             }
             else if ( "C" .equals(Grade))
             {
                GradeNum = 2;
             }
             else if ( "D" .equals(Grade))
             {
                GradeNum = 1;
             }
             else
             {
                GradeNum = 0;
             }
             Gradesum += GradeNum;
             Scoresum += Double.parseDouble(Score);
             Absencesum += Double.parseDouble(Absence);
             Timesum += Double.parseDouble(Time);
             Startsum += Double.parseDouble(Start);
             Subsum += Double.parseDouble(Sub);
             Quizsum += Double.parseDouble(Quiz);
             Ordersum += Double.parseDouble(Order);
          }
         
          idmean = idsum / (n - 1);
          Grademean = Gradesum / (n - 1);
          Scoremean = Scoresum / (n - 1);
          Absencemean = Absencesum / (n - 1);
          Timemean = Timesum / (n - 1);
          Startmean = Startsum / (n - 1);
          Submean = Subsum / (n - 1);
          Quizmean = Quizsum / (n - 1);
          Ordermean = Ordersum / (n - 1);
         
          System.out.println( "idmean: " + idmean);
          System.out.println( "Grademean: " + Grademean);
          System.out.println( "Scoremean: " + Scoremean);
          System.out.println( "Absencemean: " + Absencemean);
          System.out.println( "Timemean: " + Timemean);
          System.out.println( "Startmean: " + Startmean);
          System.out.println( "Submean: " + Submean);
          System.out.println( "Quizmean: " + Quizmean);
          System.out.println( "Ordermean: " + Ordermean);
         
          s = new Scanner(storestring.toString().trim());
          s.useDelimiter( "\t|\n" );
          idsum = 0;
          Gradesum = 0;
          Scoresum = 0;
          Absencesum = 0;
          Timesum = 0;
          Startsum = 0;
          Subsum = 0;
          Quizsum = 0;
          Ordersum = 0;
         
          while (s.hasNext())
          {
             id = s.next().trim();
             Grade = s.next().trim();
             Score = s.next().trim();
             Absence = s.next().trim();
             Time = s.next().trim();
             Start = s.next().trim();
             Sub = s.next().trim();
             Quiz = s.next().trim();
             s.useDelimiter( "\n" );
             Order = s.next().trim();
             s.useDelimiter( "\t" );
            
             double currentId = Double.parseDouble(id);
             double GradeNum = 0;
             if ( "A" .equals(Grade))
             {
                GradeNum = 4;
             }
             else if ( "B" .equals(Grade))
             {
                GradeNum = 3;
             }
             else if ( "C" .equals(Grade))
             {
                GradeNum = 2;
             }
             else if ( "D" .equals(Grade))
             {
                GradeNum = 1;
             }
             else
             {
                GradeNum = 0;
             }
            
             idsum += Math.pow(currentId - idmean, 2);
             Gradesum += Math.pow(GradeNum - Grademean, 2);
             Scoresum += Math.pow(Double.parseDouble(Score) - Scoremean, 2);

          }
         
          iddev = Math.sqrt(idsum / (n - 1));
